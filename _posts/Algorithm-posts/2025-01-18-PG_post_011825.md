```python
import heapq

def solution(scoville, K):
    answer = 0
    heapq.heapify(scoville)
    
    while scoville[0] < K:
        if len(scoville) == 1:
            return -1
        
        mild = heapq.heappop(scoville)
        n_mild = heapq.heappop(scoville)
        
        heapq.heappush(scoville, mild + (n_mild * 2))
        answer += 1
        
    return answer
```
