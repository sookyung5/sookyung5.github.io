from collections import deque 

def solution(game_board, table):
    n = len(game_board)
    
    # í¼ì¦ ëª¨ì–‘ ì¶”ì¶œ
    def get_pieces(board, target):
        visited = [[0] * n for _ in range(n)]
        result = []
        move = [[-1, 0], [0, -1], [0, 1], [1,0]]
        
        # ëŒì•„ê°€ë©´ì„œ ê³µê°„ ì°¾ê¸°
        for i in range(n):
            for j in range(n):
                if board[i][j] == target and not visited[i][j]:
                    # bfsë¡œ ì—°ê²°ëœ ë¶€ë¶„ íƒìƒ‰
                    piece = []
                    queue = deque([(i,j)])
                    visited[i][j] = 1
                    
                    while queue:
                        x, y = queue.popleft()
                        piece.append((x,y))
                        
                        for d in move:
                            next_x = x + d[0]
                            next_y = y + d[1]
                            
                            if 0 <= next_x < n and 0 <= next_y < n:
                                if board[next_x][next_y] == target and not visited[next_x][next_y]:
                                    visited[next_x][next_y] = 1
                                    queue.append((next_x, next_y))
                        
                    # ì •ê·œí™”!
                    min_x = min(x for x, y in piece)
                    min_y = min(y for x, y in piece)
                    normalized = sorted([(x - min_x, y-min_y) for x, y in piece])
                    result.append(normalized)
        return result 

    empty_pieces = get_pieces(game_board, 0)
    puzzle_pieces = get_pieces(table, 1)

    def rotate_normalize(piece):
        rotated = sorted([(y, -x) for x, y in piece])
        min_x = min(x for x,y in rotated)
        min_y = min(y for x, y in rotated)
        
        return sorted([(x - min_x, y - min_y) for x, y in rotated])
    
    # ê° ì¡°ê°ì´ ë§ëŠ”ì§€ í™•ì¸
    used_puzzle = [False] * len(puzzle_pieces)
    used_empty = [False] * len(empty_pieces)
    cnt = 0
    
    for e_idx, empty in enumerate(empty_pieces):
        matched = False
        if not used_empty[e_idx]:
            for p_idx, puzzle in enumerate(puzzle_pieces):
                if not used_puzzle[p_idx]:
                    current = puzzle
                    for _ in range(4):
                        if current == empty:
                            used_puzzle[p_idx] = True
                            cnt += len(empty)
                            matched = True
                            break
                        current = rotate_normalize(current)
                if matched:
                    break

    return cnt
