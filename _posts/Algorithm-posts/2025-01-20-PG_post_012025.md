```python
import heapq

def solution(jobs):
    jobs.sort()
    working = [] # í™
    
    cur_time = 0
    completed = 0
    total_time = 0
    
    i = 0
    while completed < len(jobs):
        while i < len(jobs) and jobs[i][0] <= cur_time:
            heapq.heappush(working, [jobs[i][1], jobs[i][0]]) 
            i += 1
        if working:
            work_time, req_time = heapq.heappop(working)
            cur_time += work_time
            total_time += (cur_time - req_time)
            completed += 1
        else:
            cur_time = jobs[i][0]
    
    return total_time // len(jobs)
```
