```python
from collections import deque

def solution(maps):
    n = len(maps)
    m = len(maps[0])

    dx = [1,-1,0,0] 
    dy = [0,0,1,-1] 

    queue = deque()
    queue.append((0,0)) 

    while queue:
        x,y = queue.popleft()

        for i in range(4):
            cur_x = x + dx[i]
            cur_y = y + dy[i]

            if cur_x < 0 or cur_x >= n or cur_y < 0 or cur_y >= m:
                continue
            if maps[cur_x][cur_y] != 1:
                continue

            maps[cur_x][cur_y] = maps[x][y] + 1
            queue.append((cur_x,cur_y))

    end = maps[n-1][m-1]
    if end == 1:
        return -1
    else:
        return end
```
